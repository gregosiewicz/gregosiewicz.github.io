import random
import matplotlib.pyplot as plt

def dist2(p, q):
    return sum((pp - qq) ** 2 for pp, qq in zip(p, q))


def center(points):
    return [sum(coord) / len(points) for coord in zip(*points)]


def assign1(points, centroids):
    assigments = []
    for p in points:
        min_dist = float('inf')
        min_idx = -1
        for i, c in enumerate(centroids):
            d = dist2(p, c)
            if d < min_dist:
                min_dist = d
                min_idx = i
        assigments.append(min_idx)
    return assigments


def assign2(points, centroids):
    assigments = []
    for p in points:
        dists = [dist2(p, c) for c in centroids]
        assigments.append(dists.index(min(dists)))
    return assigments


def assign3(points, centroids):
    return [min(range(len(centroids)), key=lambda i: dist2(p, centroids[i])) for p in points]


def calc_cc(points, assigments, k):
    clusters = [[] for _ in range(k)]
    for p, idx in zip(points, assigments):
        clusters[idx].append(p)
    centroids = []
    for cluster in clusters:
        centroids.append(center(cluster))
    return centroids, clusters


def calc_cc2(points, assigments, k):
    clusters = [[p for p, idx in zip(points, assigments) if idx == i] for i in range(k)]
    centroids = [center(c) for c in clusters]
    return centroids, clusters


def kmeans(points, k, max_iter, eps):
    centroids = random.sample(points, k)

    for _ in range(max_iter):
        assigments = assign1(points, centroids)
        new_centroids, clusters = calc_cc2(points, assigments, k)

        d = max(dist2(c1, c2) for c1, c2 in zip(centroids, new_centroids))
        centroids = new_centroids
        if d < eps:
            break

    return centroids, clusters


random.seed(13)
points = [[random.uniform(0, 10), random.uniform(0, 10)] for _ in range(1000)]

centroids, clusters = kmeans(points, 5, 100, 1e-4)

print("Centroids:", [[f"{coord:.2f}" for coord in c] for c in centroids])
print("Cluster sizes:", [len(c) for c in clusters])


colors = ['red', 'blue', 'green', 'purple', 'orange', 'yellow']

for i, cluster in enumerate(clusters):
    if cluster:
        x = [p[0] for p in cluster]
        y = [p[1] for p in cluster]
        plt.scatter(x, y, color=colors[i % len(colors)], alpha=0.6, label=f'Cluster {i+1}')

cent_x = [c[0] for c in centroids]
cent_y = [c[1] for c in centroids]
plt.scatter(cent_x, cent_y, color='black', marker='x', s=100, label='Centroids')
